class Point:

  def __init__(self, x, y, z=0):
    self.x = x
    self.y = y
    self.z = z

  def __repr__(self):
    return "( " + str(self.x) + ", " + str(self.y) + " )"
  def __add__(self, b):
    return Point(self.x+b.x, self.y+b.y)
  def __sub__(self, b):
    return Point(self.x-b.x, self.y-b.y)
  def __mul__(self, b):
    return Point(b*self.x, b*self.y)
  __rmul__ = __mul__
    
  def IsIn(self, t):
    ''' Verifica se p est em t usando o baricentro'''
    b = t.v[1] - t.v[0]
    c = t.v[2] - t.v[0]
    d = self   - t.v[0]
    det = c.x*b.y-c.y*b.x;
    u = (d.x*b.y-d.y*b.x)//float(det);
    v = (c.x*d.y-c.y*d.x)//float(det);
    return u >= 0 and v >= 0 and u + v < 1

import numpy as np
from math import sqrt

class TriangulacaoDelaunay:

    def __init__(self):

        self.load_coords_default()

        self.triangles = {}
        self.circles = {}

        Triang1 = (0, 1, 3)
        Triang2 = (2, 3, 1)
        self.triangles[Triang1] = [Triang2, None, None]
        self.triangles[Triang2] = [Triang1, None, None]

        for t in self.triangles:
            self.circles[t] = self.circuncentro(t)

    def runDelaunay(self, point):

        point = np.asarray(point)
        idx = len(self.coords)
        self.coords.append(point)

        bad_triangles = []
        for T in self.triangles:
            centro, raio = self.circles[T]
            if np.sum(np.square(centro - point)) <= raio:
                bad_triangles.append(T)
        boundary = []
        T = bad_triangles[0]
        edge = 0
        while True:
            tri_op = self.triangles[T][edge]
            if tri_op not in bad_triangles:
                boundary.append((T[(edge+1) % 3], T[(edge-1) % 3], tri_op))
                edge = (edge + 1) % 3
                if boundary[0][0] == boundary[-1][1]:
                    break
            else:
                edge = (self.triangles[tri_op].index(T) + 1) % 3
                T = tri_op

        for T in bad_triangles:
            del self.triangles[T]
            del self.circles[T]

        new_triangles = []
        for (edge0, edge1, tri_op) in boundary:
            T = (idx, edge0, edge1)
            self.circles[T] = self.circuncentro(T)
            self.triangles[T] = [tri_op, None, None]

            if tri_op:
                for i, neigh in enumerate(self.triangles[tri_op]):
                    if neigh:
                        if edge1 in neigh and edge0 in neigh:
                            self.triangles[tri_op][i] = T

            new_triangles.append(T)

        N = len(new_triangles)
        for i, T in enumerate(new_triangles):
            self.triangles[T][1] = new_triangles[(i+1) % N]
            self.triangles[T][2] = new_triangles[(i-1) % N]

    def getDelaunayTriangulation(self):
        coord = self.coords[4:]
        tris = [(a-4, b-4, c-4)
                for (a, b, c) in self.triangles if a > 3 and b > 3 and c > 3]
        return coord, tris
    
    def circuncentro(self, triangles):

        points = np.asarray([self.coords[v] for v in triangles])
        points2 = np.dot(points, points.T)
        A = np.bmat([[2 * points2,[[1],[1],[1]]], [[[1, 1, 1, 0]]]])

        b = np.hstack((np.sum(points * points, axis=1), [1]))
        x = np.linalg.solve(A, b)
        baricentro_coords = x[:-1]
        centro = np.dot(baricentro_coords, points)

        raio = np.sum(np.square(points[0] - centro))
        return (centro, raio)

    def load_coords_default(self):
        centro = np.asarray((0,0))
        raio = 999999999

        # Cria os quatro pontos extremos do quadrado
        self.coords = [ centro + raio * np.array((-1, -1)),
                        centro + raio * np.array((+1, -1)),
                        centro + raio * np.array((+1, +1)),
                        centro + raio * np.array((-1, +1))]

def plot_triangulacao(df):

  xs = df['x'].tolist()
  ys = df['y'].tolist() 

  delaunay = TriangulacaoDelaunay()

  for i in range(len(xs)):
    delaunay.runDelaunay((xs[i],ys[i]))

  points, triangles = delaunay.getDelaunayTriangulation()
  
  tri = Triangulation(xs,ys,triangles)

  plt.triplot(tri, color='red')
  plt.scatter(xs,ys)
  plt.show()

  return triangles
